package com.syf.payfone.model;

import java.util.Map;

/**
 * Base interface for Excel records following Interface Segregation Principle
 */
public interface ExcelRecord {
    String getPhoneNumber();
    String getLineType();
    String getConfirmed();
    String getSsn();
    String getDob();
    String getChannelId();
    String getTransactionId();
    String getStatus();
    void setTransactionId(String transactionId);
    void setStatus(String status);
    
    /**
     * Get the identifier field value (AccountNumber or CIF)
     */
    String getIdentifier();
    
    /**
     * Convert to map for easy processing
     */
    Map<String, String> toMap();
    
    /**
     * Validate mandatory fields
     */
    boolean isValid();
    
    /**
     * Get validation error message
     */
    String getValidationError();
}
